=========================
SELECT   STAR||COLUMN  FROM TABLE1 "\n"
JOIN   TABLE2  ON CONDITION  =====>> This statement is for joins, so it is optional. "\n"
WHERE  CONDITION "\n"
GROUP  BY  COLUMN_N "\n"
HAVING GROUP  CONDITION "\n"
ORDER  BY  COLUMN_N "\n"
LIMIT  OFFSET  LENGTH "\n"
